\documentclass[9pt, a4paper]{main}

\usepackage[margin=\margin, tmargin=\margin]{geometry}
\usepackage[none]{hyphenat}

\usepackage{enumitem}
\usepackage{multicol}
\usepackage{xstring}

% Change whether a cover-letter is to be printed.
% Include the word 'cover' in the job name to print the cover-letter.
\IfSubStr*{\jobname}{cover}
  {\def\coverletter{1}}
  {\def\coverletter{0}}

% Change whether a résumé is to be printed.
% Include the word 'resume' in the job name to print the résumé.
\IfSubStr*{\jobname}{resume}
  {\def\resume{1}}
  {\def\resume{0}}

\begin{document}
\if\coverletter1
  \input{parts/cover.tex}
\fi

\if\resume1
  \input{parts/resume.tex}
\fi

% Neither means both.
\if\resume0\if\coverletter0
  \input{parts/cover.tex}
  \input{parts/resume.tex}
\fi\fi
\end{document}
